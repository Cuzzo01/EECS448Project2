recursion Reveal pseudocode

recursionReveal (curX, curY)
{
	if (cell has not been revealed)
	{
		if (cell contains a number)
		{
			reveal cell
		}
		if (cell is empty)
		{
			reveal cell
			for (let x = -1; x <= 1; x++)
			{
				for (let y = -1; y <= 1; y++)
				{
					recursionReveal (curX + x, curY + y)
				}
			}
		}
	}
}
