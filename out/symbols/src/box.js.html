  5 /**
  6 * Class representing a box that may or may not contain a mine or flag
  7 * @name Box
  8 * @constructor
  9 * @classdesc Class representing a box that may or may not contain a mine or flag
 10 */
 11 class Box {
 12   /**
 13   * @param  {number} i - The row index for the Box
 14   * @param  {number} j - The col index for the Box
 15   * @param  {number} w - The pixel width of the Div element used to display the Box
 16   * @param  {number} boom - Value that is -1 is there is a mine, or 0 - 8 depending on surrounding tiles
 17   */
 18   constructor(i, j, w, boom) {
 19     this.div = document.createElement('div');
 20     this.div.classList.add('box');
 21     this.div.style.gridArea = [(i+1).toString(),(j+1).toString(), 'span 1', 'span 1'].join(' / ') ;
 22     this.div.setAttribute('onmousedown','mouseDown('+i+','+j+',event)');
 23     this.i = i;
 24     this.j = j;
 25     this.w = w
 26     this.boom = 0
 27     this.revealed = false
 28     this.flagged = false
 29     this.bufferBox = null;
 30     this.tempBox = null;
 31     this.origin = null;
 32     this.cheat = false
 33   }
 34 
 35   /**
 36   * @method attach
 37   * @param {HTMLElement} Board an HTML element that will become the parent for this Box
 38   */
 39   attach(board) {
 40     board.appendChild(this.div);
 41   }
 42 
 43   /**
 44   * @method generateBufferBox create a buffer box on mine tiles for runAwayMode
 45   */
 46   generateBufferBox() {
 47     this.div.classList.add('mine');
 48     let rect = this.div.getBoundingClientRect();
 49     this.origin = [(rect.left+10), (rect.top+10)];
 50     this.bufferBox = document.createElement('div');
 51     this.bufferBox.setAttribute("data-originX", rect.left + window.pageXOffset);
 52     this.bufferBox.setAttribute("data-originY", rect.top + window.pageYOffset);
 53     this.bufferBox.classList.add('bufferBox');
 54     document.body.appendChild(this.div);
 55     this.tempBox = document.createElement('div');
 56     this.tempBox.classList.add('tempBox');
 57     this.tempBox.style.gridArea = [(this.i+1).toString(),(this.j+1).toString(), 'span 1', 'span 1'].join(' / ') ;
 58     document.getElementById('board').appendChild(this.tempBox);
 59     this.div.style.position = 'absolute';
 60     this.div.style.top = this.bufferBox.getAttribute('data-originY') + 'px';
 61     this.div.style.left = this.bufferBox.getAttribute('data-originX') + 'px'
 62     this.div.appendChild(this.bufferBox);
 63   }
 64 
 65   /**
 66    * @function deleteBufferBox removes the DOM element for the box used to handle runAwayMode
 67    */
 68   deleteBufferBox() {
 69     this.div.classList.remove('mine');
 70     this.div.removeChild(this.bufferBox);
 71     document.getElementById('board').appendChild(this.div);
 72     document.getElementById('board').removeChild(this.tempBox);
 73     this.div.style.backgroundColor = null;
 74     this.div.style.borderBottom = null;
 75     this.div.style.position = null;
 76     this.div.style.transition = null;
 77     this.div.style.top = null;
 78     this.div.style.left = null;
 79     this.bufferBox = null;
 80   }
 81 
 82   /**
 83    * @function runAway event handler to update Box positions and color in runAway mode.
 84    * @param {GlobalEvent} Event passed by callback to obtain mouse position for calculations
 85    */
 86   runAway(e) {
 87     let rect = this.div.getBoundingClientRect();
 88     let left = rect.left + window.pageXOffset;
 89     let top = rect.top + window.pageYOffset;
 90     let xpos = e.pageX;
 91     let ypos = e.pageY;
 92     let distanceFromOrigin = Math.sqrt(Math.pow(left-this.bufferBox.getAttribute('data-originX'),2) + Math.pow(top-this.bufferBox.getAttribute('data-originY'),2));
 93     let redValue = clamp((distanceFromOrigin)/20,0,1)*148;
 94     if(!this.flagged)
 95       this.div.style.backgroundColor = `rgb(${redValue+107},${220-redValue},${254-redValue})`;
 96     else
 97       this.div.style.borderBottom = `18px inset rgb(${redValue+107},${220-redValue},${254-redValue})`;
 98     let direction = [(left+10) - xpos, (top+10) - ypos];
 99     let magnitude = Math.sqrt(Math.pow(direction[0],2) + Math.pow(direction[1],2));
100     if(magnitude<=22){
101       direction = [direction[0]/magnitude,direction[1]/magnitude];
102       let amplitude = 1/magnitude;
103       if(amplitude<0.046)
104         amplitude = 0;
105       let newPosition = [left +  amplitude * 30 * direction[0], top +  amplitude * 30 * direction[1]];
106       this.div.style.transition = 'all 0s';
107       this.div.style.left = newPosition[0] + 'px';
108       this.div.style.top = newPosition[1] + 'px';
109       return true;
110     }
111     else{
112       this.div.style.transition = 'all 1s';
113       this.div.style.top = this.bufferBox.getAttribute('data-originY') + 'px';
114       this.div.style.left = this.bufferBox.getAttribute('data-originX') + 'px';
115       return false;
116     }
117   }
118 
119   /**
120    * @function toggleCheat Change 'cheat' state, and handle DOM manipulation to update rendering
121    */
122   toggleCheat() {
123     this.cheat = !this.cheat;
124     if (this.cheat) {
125       if(this.boom == -1) {
126         this.div.classList.add("revealedMine");
127       } else {
128         this.div.classList.add("revealed");
129         if(this.boom != 0)
130           this.div.innerHTML = this.boom;
131       }
132     } else {
133       if(this.boom == -1) {
134         this.div.classList.toggle("revealedMine");
135       } else {
136         if (!this.revealed) {
137           this.div.classList.toggle("revealed");
138           if(this.boom != 0)
139             this.div.innerHTML = "";
140         }
141       }
142     }
143   }
144 
145   /**Changes Box to the Revealed state
146     @function reveal Uncovers Box. If Box is a mine, trigger game end. If box
147         has proximity count of 0, trigger recursion.
148     */
149   reveal() {
150     this.revealed = true;
151     if(this.flagged) {
152       this.toggleFlag();
153     }
154     if(this.boom == -1) {
155       this.div.classList.add("revealedMine");
156     } else {
157       this.div.classList.add("revealed");
158       if(this.boom != 0)
159         this.div.innerHTML = this.boom;
160     }
161     if(this.boom == 0 && !this.flagged) {
162       this.recurseReveal();
163     } else if(this.boom == -1 && !this.flagged) {
164       endGameLose();
165     }
166   }
167 
168   /**Recursive reveal call
169   * @function reveal calls reveal on all surrounding boxes
170   */
171   recurseReveal() {
172     for( let i = -1; i <= 1; i++ ) {
173       for( let j = -1; j <= 1; j++ ) {
174         let newI = this.i + i
175         let newJ = this.j + j
176         if (newI >= 0 && newI < gameBoard.length) {
177           if (newJ >= 0 && newJ < gameBoard[newI].length) {
178             if(!gameBoard[newI][newJ].revealed && !gameBoard[newI][newJ].flagged)
179               gameBoard[newI][newJ].reveal();
180           }
181         }
182       }
183     }
184   }
185 
186   /**Place or remove a flag
187   *@function endGameWin changes the flag state of the Box
188   */
189   toggleFlag() {
190     this.flagged = !this.flagged;
191     this.div.classList.toggle("flagged");
192   }
193 }
194 function clamp(num, min, max) {
195   return num <= min ? min : num >= max ? max : num;
