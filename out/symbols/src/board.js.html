 18 /**Toggles global boolean runAwayMode, and depending on its new value start a
 19   * looped interval that fires a runAway() event from each Box object every 100 ms
 20   * A reference to this interval is saved as global variable "interval".
 21   * if runAwayMode is now true. if false, it clears the global interval.
 22   * @returns Nothing
 23   */
 24 function toggleRunAwayMode() {
 25   runAwayMode = !runAwayMode;
 26   if(runAwayMode){
 27     let runAwayText = document.createElement('div');
 28     runAwayText.id = 'runAwayText';
 29     runAwayText.innerHTML = 'Run Away Mode Enabled!';
 30     document.body.appendChild(runAwayText);
 31     interval = setInterval(function(){
 32       for(let i=0;i<rows;i++){
 33         for(let j=0;j<cols;j++){
 34             if(gameBoard[i][j].boom == -1){
 35               gameBoard[i][j].runAway(globalEvent);
 36             }
 37           }
 38       }
 39     }, 100);
 40   }
 41   else{
 42     document.body.removeChild(document.getElementById('runAwayText'));
 43     clearInterval(interval);
 44   }
 45   for(let i=0;i<rows;i++){
 46     for(let j=0;j<cols;j++){
 47         if(gameBoard[i][j].boom == -1)
 48           if(runAwayMode){
 49             gameBoard[i][j].generateBufferBox();
 50           }
 51           else{
 52             gameBoard[i][j].deleteBufferBox();
 53           }
 54     }
 55   }
 56 }

 68 /**Takes input from HTML to construct a 2D array of Box objects along with their
 69 * related div HTML tag.
 70 * @returns Nothing
 71 */
 72 function setup() {
 73   if(runAwayMode)
 74     toggleRunAwayMode();
 75   var inputs = Array.from(document.getElementsByClassName("setupInput")); // Input elements
 76   if( inputs.some( (x) => !x.reportValidity() ) ) { return; }             // Stop if invalid
 77   [ rows, cols, mineCount ] = inputs.map( (x) => Number(x.value) );   // Assign variables
 78   [ width, height ] = [ rows * w + 1,                                 // Get width/height in px
 79                         cols * w + 1 ];
 80 
 81   if (document.contains(document.getElementById('board'))) {
 82     document.getElementById('board').remove();
 83   }
 84   let board = document.createElement('div');
 85   board.id = 'board';
 86   board.addEventListener( "contextmenu", (evt) => evt.preventDefault() );
 87   endGameCheck = false;
 88   gameBoard = build2DArray(rows, cols);
 89   for (var i = 0; i < rows; i++) {
 90     for (var j = 0; j < cols; j++) {
 91       let tempBox = new Box(i, j, w, 0);
 92       tempBox.attach(board);
 93       gameBoard[i][j] = tempBox;
 94     }
 95   }
 96   document.getElementById('boardDiv').appendChild(board);
 97   document.body.onmousemove = function(e)
 98   {
 99     globalEvent = e;
100     if(runAwayMode){
101       for(let i=0;i<rows;i++){
102         for(let j=0;j<cols;j++){
103             if(gameBoard[i][j].boom == -1){
104               gameBoard[i][j].runAway(globalEvent);
105               }
106             }
107         }
108       }
109   };
110   //initBoom
111   while( mineCount != 0 ) {
112     var randX = Math.floor( Math.random() * rows );
113     var randY = Math.floor( Math.random() * cols );
114     if( gameBoard[randX][randY].boom == 0 ) {     // consider renaming 'boom' to something more descriptive
115       gameBoard[randX][randY].boom = -1;
116       mineCount--;
117     }
118   }
119   //setup value
120   for( var i = 0 ; i < rows; i++ ) {
121     for( var j = 0; j < cols ; j ++ ) {
122       // should we have 'boom' hold both count and is-a-mine info? it makes sense.
123       gameBoard[i][j].boom = gameBoard[i][j].boom ? -1 : getCenterCount( i, j );
124     }
125   }
126 }

128 /** Function callback for handling mouse clicks
129   * @function mouseDown Callback for mouse events
130   * @param {number} row of the clicked box
131   * @param {number} col of the clicked box
132   * @param {Event} Event the event object passed to this callback
133   * @returns Nothing
134 */
135 function mouseDown(i, j, e) {
136   // row should be y-related, col x-related. doesn't work that way rn
137   // This switch statement changes controls to use left and right mouse buttons,
138   // no more button pressing
139   if(e.target !== e.currentTarget)
140     return;
141   e = e || window.event;
142   let mouseButton;
143   if ("which" in e)  // Gecko (Firefox), WebKit (Safari/Chrome) & Opera
144       mouseButton = e.which;
145   switch( mouseButton ) {
146     case 1:
147       gameBoard[i][j].reveal();
148       checkWinConditions();
149       break;
150     case 3:
151       gameBoard[i][j].toggleFlag();
152       break;
153     default:
154       break;
155   }
156 }

158 /**
159   * Determines if the gameboard is in a valid win state
160   * @returns Nothing
161   */
162 function checkWinConditions() {
163   var gameWon = !gameBoard.some( ( row ) => {               // there are no rows such that
164     return row.some( (cell) => {                                  // the row has no cell such that
165       return ( !cell.revealed && ( cell.boom != -1 ) );    // that cell is not revealed and not a bomb
166     })
167   });
168   if( gameWon ) {
169     endGameWin();
170   }
171 }

173 /** Reveal all tiles and send win message
174   * @returns Nothing
175   */
176 function endGameWin() {
177   if( endGameCheck == false ) {
178     endGameCheck = true;
179     for( let i = 0; i < rows; i++ ) {
180       for( let j = 0; j < cols; j++ ) {
181         if( !gameBoard[i][j].revealed && !gameBoard[i][j].flagged ) {
182           gameBoard[i][j].reveal();
183         }
184       }
185     }
186     alert("Mines successfully swept");  // I removed the timeout. The time out allows the board to finish rerendering the board.
187   }
188 }

233 /**
234   * Returns the number of mines surrounding Box at (x, y)
235   * @function getCenterCount returns the number of mines surrounding a Box
236   * @param {number} x row index of Box
237   * @param {number} y column index of Box
238   * @returns the number of mines
239   */
240 function getCenterCount( x , y ) {
241   var position = [
242     [ x - 1, y - 1 ],
243     [ x - 1, y     ],
244     [ x - 1, y + 1 ],
245     [ x    , y - 1 ],
246     [ x    , y + 1 ],
247     [ x + 1, y - 1 ],
248     [ x + 1, y     ],
249     [ x + 1, y + 1 ],
250   ];
251   var count = 0;
252   for(var i = 0 ; i < position.length ; i++) {
253     var row = position[i][0];
254     var col = position[i][1];
255     if( gameBoard[ row ] != undefined && gameBoard[ row ][ col ] != undefined ) {
256       count += (gameBoard[ row ][ col ].boom == -1);
257     }
258   }
259   return count;
260 }
